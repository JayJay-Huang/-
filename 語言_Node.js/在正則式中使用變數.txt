function HtmlNormalize(pContentsStr){
    pContentsStr
    .replace(/<([a-z][a-z0-9]*)[^>]*(\/?)>/si,'<$1$2>')
    // .replace(/<(p[a-z0-9]*)[^>]*(\/?)>/si,'<$1$2>')
    // .replace(/<(strong[a-z0-9]*)[^>]*(\/?)>/si,'<$1$2>');
    let response = pContentsStr;
    return response;
};
